367  // creating a file and a DataOutputStream to the file
368  /*
369  Use BufferedOutputStream to increase write speed:
370  */
371  var dos = new DataOutputStream (file.append_to (FileCreateFlags.NONE));
372  // writing a short string to the stream
373  // dos.put_string ("this is the first line\n");
374  // string text = "this is the second line\n";
375  // For long string writes, a loop should be used, because sometimes not all data can be written in one run
376  // 'written' is used to check how much of the string has already been written
377  uint8[] data = text.data;
378  long written = 0;
379  while (written < data.length) {
380  // sum of the bytes of 'text' that already have been written to the stream
381  written += dos.write (data[written:data.length]);
382  }
383  }
